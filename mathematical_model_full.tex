\section{Sets}
\begin{itemize}
    \item $DC$: set of candidate distribution centers.
    
    \item $DS$: set of candidate dark stores.
    
    \item $MDCP$: set of candidate parking locations for mobile distribution centers.
    
    \item $F = DC \cup DS \cup MDCP$: set of all candidate facilities in the network.
    
    \item $P$: set of ports:
    \[
    P = \{\text{San Antonio}, \text{Valparaíso}\}
    \]
    
    \item $A$: set of airports:
    \[
    A = \{\text{Arturo Merino Benítez International Airport}\}
    \]

    \item $E = P \cup A$: set of international entry points to Chile.
    
    \item $WM$: set of wholesale markets:
    \[
    WM = \{\text{La Vega Central}, \text{Lo Valledor}\}
    \]
    
    \item $S^{imp}$: set of imported supplier countries:
    \[
    S^{imp} = \{\mathrm{US}, \mathrm{Mexico}, \mathrm{Peru}, \mathrm{Ecuador}, \mathrm{Bolivia}, \mathrm{Argentina}, \mathrm{Spain}\}
    \]

    \item $S^{cl}$: set of domestic supplier regions in Chile:
    \[
    S^{cl} = \{\mathrm{Chile\text{-}Scl}, \mathrm{Chile\text{-}Vap}, \mathrm{Chile\text{-}Maul}, \mathrm{Chile\text{-}Coq}\}
    \]

    \item $S = S^{imp} \cup S^{cl}$: set of all supplier origins, where
    \[
    S^{imp} \cap S^{cl} = \varnothing
    \]

    \item $K$: set of products:
    \[
    K = \{\text{Roma Tomatoes}, \text{Cauliflower}, \text{Broccoli}, \text{Asparagus}, \text{Green Bell Pepper}\}
    \]
    
    \item $T$: set of weeks in the planning horizon:
    \[
    T = \{1, 2, \dots, 52\}
    \]
    
    \item $R$: set of transportation modes:
    \[
    R = \{\text{sea}, \text{air}, \text{truck}\}
    \]

    \item $\mathcal{Z}$: set of all feasible facility configurations, where each $z \in \mathcal{Z}$ is a binary vector in $\{0,1\}^{|F|}$, excluding the null configuration.

\end{itemize}

\subsection{Distance Parameters}

\begin{itemize}
    \item $D^{DC,DS}_{i,j}$: distance between distribution center $i \in DC$ and dark store $j \in DS$, expressed in km.

    \item $D^{DC,MDCP}_{i,m}$: distance between distribution center $i \in DC$ and mobile distribution center parking location $m \in MDCP$, expressed in km.

    \item $D^{E,DC}_{e,i}$: distance between international entry point $e \in E$ and distribution center $i \in DC$, expressed in km.

    \item $D^{S^{cl},DC}_{s,i}$: distance between domestic supplier $s \in S^{cl}$ and distribution center $i \in DC$, expressed in km.

    \item $D^{MDCP,MDCP}_{m,n}$: distance between mobile distribution center parking locations $m \in MDCP$ and $n \in MDCP$, expressed in km.
\end{itemize}

\subsection{Time Parameters}

\begin{itemize}
    \item $T^{DC,DS}_{i,j}$: travel time between distribution center $i \in DC$ and dark store $j \in DS$, expressed in hours.

    \item $T^{DC,MDCP}_{i,m}$: travel time between distribution center $i \in DC$ and mobile distribution center parking location $m \in MDCP$, expressed in hours.

    \item $T^{P,DC}_{p,i}$: travel time between port $p \in P$ and distribution center $i \in DC$, expressed in hours.

    \item $T^{A,DC}_{a,i}$: travel time between airport $a \in A$ and distribution center $i \in DC$, expressed in hours.
    
    \item $T^{E,DC}_{e,i}$: travel time between international entry point $e \in E$ and distribution center $i \in DC$, expressed in hours.
    
    \item $T^{S^{cl},DC}_{s,i}$: travel time between domestic supplier $s \in S^{cl}$ and distribution center $i \in DC$, expressed in hours.

    \item $T^{MDCP,MDCP}_{m,n}$: travel time between mobile distribution center parking locations $m \in MDCP$ and $n \in MDCP$, expressed in hours.
    
    \item $T^{S^{imp},E}_{s,e}$: travel time between imported supplier $s \in S^{imp}$ and international entry point $e \in E$, expressed in hours.
\end{itemize}
