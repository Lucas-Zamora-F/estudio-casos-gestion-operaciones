\section{Model Assumptions}

\begin{itemize}
    \item Demand for each product is assumed to be constant across all time periods.
    
    \item Households are assumed to be homogeneous in terms of consumption rates and ordering behavior.
    
    \item Each household places exactly one order per week, consistently across all weeks.
    
    \item All transportation and replenishment processes are assumed to be instantaneous. Therefore, products purchased from any supplier are assumed to arrive immediately at the corresponding port, airport, or distribution center, and any subsequent transfers within the network are also assumed to occur without delay.

    \item Last-mile delivery distances are approximated using an average distance from each facility type to the households within its coverage area.
    
    \item Households are assumed to be uniformly distributed within the coverage area of each facility.
    
    \item Delivery routes are not explicitly modeled; instead, transportation costs are approximated using average distances.
    
\end{itemize}

\section{Sets}
\begin{itemize}
    \item $CD$: set of candidate distribution centers.
    
    \item $DS$: set of candidate dark stores.
    
    \item $MDCP$: set of candidate parking locations for mobile distribution centers.
    
    \item $F = CD \cup DS \cup MDCP$: set of all candidate facilities in the network.
    
    \item $P$: set of ports:
    \[
    P = \{\text{San Antonio}, \text{Valparaíso}\}
    \]
    
    \item $A$: set of airports:
    \[
    A = \{\text{Arturo Merino Benítez International Airport}\}
    \]
    
    \item $WM$: set of wholesale markets:
    \[
    WM = \{\text{La Vega Central}, \text{Lo Valledor}\}
    \]
    
    \item $S$: set of supplier countries:
    \[
    \begin{aligned}
    S = \{ & \mathrm{US}, \mathrm{Mexico}, \mathrm{Peru}, \mathrm{Ecuador}, \\
        & \mathrm{Bolivia}, \mathrm{Argentina}, \mathrm{Spain}, \\
        & \mathrm{Chile\text{-}Scl}, \mathrm{Chile\text{-}Vap}, \\
        & \mathrm{Chile\text{-}Maul}, \mathrm{Chile\text{-}Coq} \}
    \end{aligned}
    \]

    \item $K$: set of products:
    \[
    K = \{\text{Roma Tomatoes}, \text{Cauliflower}, \text{Broccoli}, \text{Asparagus}, \text{Green Bell Pepper}\}
    \]
    
    \item $T$: set of weeks in the planning horizon:
    \[
    T = \{1, 2, \dots, 52\}
    \]
    
    \item $R$: set of transportation modes:
    \[
    R = \{\text{sea}, \text{air}, \text{truck}\}
    \]

    \item $\mathcal{Z}$: set of all feasible facility configurations, where each $z \in \mathcal{Z}$ is a binary vector in $\{0,1\}^{|F|}$, excluding the null configuration.

    \item $I \subseteq \mathcal{Z}$: set of singleton facility configurations, where each $i \in I$ is a binary vector with exactly one component equal to 1 and all others equal to 0.
    

\end{itemize}

\section{Parameters}

% =========================================
\subsection{Distance Parameters}

All distances are measured in kilometers (km).

\begin{itemize}
    \item $d^{CD,DS}_{ij}$: distance from distribution center $i \in CD$ to dark store $j \in DS$.

    \item $d^{MDCP}_{ij}$: distance between mobile distribution center parking locations $i,j \in MDCP$.

    \item $d^{CD,P}_{ij}$: distance from distribution center $i \in CD$ to port $j \in P$.

    \item $d^{CD,A}_{ij}$: distance from distribution center $i \in CD$ to airport $j \in A$.

    \item $d^{CD,WM}_{ij}$: distance from distribution center $i \in CD$ to wholesale market $j \in WM$.
    
    \item $d^{S_{cl},CD}_{ij}$: distance from Chilean supplier $i \in S_{cl}$ to distribution center $j \in CD$ (direct transport).
    
    \item $\bar{d}_f$: average distance (km) from facility $f \in F$ to the households it serves.
    
\end{itemize}

% =========================================
\subsection{Coverage Parameters}

\begin{itemize}
    \item $z \in \{0,1\}^{|F|}$: binary vector representing a facility configuration, where each component indicates whether a specific facility is open (1) or closed (0). The components of $z$ are ordered such that the first $|CD|$ elements correspond to distribution centers (CD), the next $|DS|$ elements correspond to dark stores (DS), and the last $|MDCP|$ elements correspond to mobile distribution centers (MDCP).
    
    \item $p(z)$: number of households that can be covered under facility configuration $z$, for all $z \in \mathcal{Z}$.
    
    \item $P_i = p(i)$: maximum number of households that can be covered by singleton configuration $i \in I$.
    
\end{itemize}

% =========================================
\subsection{Cost Parameters}

\begin{itemize}
    \item $c^{buy}_{k,s,r}$: purchase cost (USD/kg) of product $k \in K$ from supplier $s \in S$ using transport mode $r \in R$.

    \item $c^{WM}_{k,w}$: purchase cost (USD/kg) of product $k \in K$ at wholesale market $w \in WM$.

    \item $v_k$: selling price (USD/kg) of product $k \in K$.
    
    \item $c^{tr}$: transportation cost per kg per km by truck (USD/kg/km).
    
    \item $c^{del}$: last-mile delivery cost per kg per km (USD/kg/km).
    
    \item $c^{mov}$: movement cost per kilometer (USD/km) for the mobile distribution center.

\end{itemize}

% =========================================
\subsection{Availability Parameters}

\begin{itemize}
    \item $a_{k,s,t}$: binary parameter equal to 1 if product $k \in K$ is available from supplier $s \in S$ in period $t \in T$.

    \item $g_{k,r}$: binary parameter equal to 1 if product $k \in K$ can be transported using mode $r \in R$.

    \item $u^{P}_{s,p}$: binary parameter equal to 1 if supplier $s \in S$ can ship through port $p \in P$.

    \item $u^{A}_{s,a}$: binary parameter equal to 1 if supplier $s \in S$ can ship through airport $a \in A$.
\end{itemize}

% =========================================
\subsection{Transportation Parameters}

\begin{itemize}

    \item $t^{first}_{s,p,r}$: transportation time (hours) from supplier $s \in S$ to entry point $p \in P \cup A$ using mode $r \in R$. 
    For domestic suppliers ($s \in$ Chile), this value is equal to 0.

    \item $\alpha^{LH}_{r}$: long-haul travel time coefficient (hours/km) for transportation mode $r \in R$.

    \item $t^{LH}_{i,j,r}$: long-haul transportation time (hours) between locations $i$ and $j$, computed as:
    \[
    t^{LH}_{i,j,r} = \alpha^{LH}_{r} \cdot d_{i,j}
    \]

    \item $\alpha^{city}$: urban travel time coefficient (hours/km) for local distribution.

    \item $t^{city}_{i,j}$: urban transportation time (hours) between facilities or to customers, computed as:
    \[
    t^{city}_{i,j} = \alpha^{city} \cdot d_{i,j}
    \]

    \item $g(z)$: minimum total travel distance (km) required for the mobile distribution center to visit all active parking locations in configuration $z \in \mathcal{Z}$ and return to its starting point.

\end{itemize}

% =========================================
\subsection{Demand Parameters}

\begin{itemize}
    \item $q_k$: demand rate (kg/person/week) of product $k \in K$.
    
    \item $\bar{h}$: average number of people per household (persons/household).
    
    \item $f^{del}$: flat delivery fee per order (USD/order), equal to 2.5 USD.
\end{itemize}
\newpage

\section{Decision Variables}

\begin{itemize}

    \item $x_z \in \{0,1\}$: equals 1 if facility configuration $z \in \mathcal{Z}$ is selected, and 0 otherwise.

    \item $H_t$: number of households served in week $t \in T$.
    
    \item $x_{k,s,t} \geq 0$: quantity (kg) of product $k \in K$ purchased from supplier $s \in S$ in period $t \in T$.
    
    \item $x_{k,w,t} \geq 0$: quantity (kg) of product $k \in K$ purchased from wholesale market $w \in WM$ in period $t \in T$.
    
    \item $y^{P}_{k,p,i,t} \geq 0$: quantity (kg) of product $k \in K$ transported from port $p \in P$ to distribution center $i \in CD$ in period $t \in T$.
    
    \item $y^{A}_{k,a,i,t} \geq 0$: quantity (kg) of product $k \in K$ transported from airport $a \in A$ to distribution center $i \in CD$ in period $t \in T$.
    
    \item $y^{S}_{k,s,i,t} \geq 0$: quantity (kg) of product $k \in K$ transported from Chilean supplier $s \in S_{cl}$ to distribution center $i \in CD$ in period $t \in T$.
    
    \item $y^{WM}_{k,w,i,t} \geq 0$: quantity (kg) of product $k \in K$ transported from wholesale market $w \in WM$ to distribution center $i \in CD$ in period $t \in T$.
    
    \item $y^{CD,DS}_{k,i,j,t} \geq 0$: quantity (kg) of product $k \in K$ transported from distribution center $i \in CD$ to dark store $j \in DS$ in period $t \in T$.
    
    \item $h_{f,t} \geq 0$: number of households served by facility $f \in F$ in period $t \in T$.
    
    
\end{itemize}

\section{Objective Function}

The objective is to maximize total profit over the planning horizon, defined as total revenue minus all relevant costs.

\vspace{0.3cm}

\textbf{Revenue:}
\[
\sum_{t \in T} 
\left[
    H_t \cdot 
    \left(
        \bar{h} \cdot \sum_{k \in K} q_k \cdot v_k
        + f^{del}
    \right)
\right]
\]

\vspace{0.3cm}

\textbf{Purchasing Cost:}
\[
\sum_{t \in T}
\left[
    \sum_{k \in K} \sum_{s \in S} c^{buy}_{k,s} \, x_{k,s,t}
    +
    \sum_{k \in K} \sum_{w \in WM} c^{WM}_{k,w} \, x_{k,w,t}
\right]
\]

\vspace{0.3cm}

\textbf{Long Haul Cost:}
\[
\sum_{t \in T}
\Bigg[
    \sum_{k \in K} \sum_{p \in P} \sum_{i \in CD} 
    c^{tr} \cdot d^{CD,P}_{i,p} \cdot y^{P}_{k,p,i,t}
    +
    \sum_{k \in K} \sum_{a \in A} \sum_{i \in CD} 
    c^{tr} \cdot d^{CD,A}_{i,a} \cdot y^{A}_{k,a,i,t}
\]
\[
    +
    \sum_{k \in K} \sum_{s \in S_{cl}} \sum_{i \in CD} 
    c^{tr} \cdot d^{S_{cl},CD}_{s,i} \cdot y^{S}_{k,s,i,t}
    +
    \sum_{k \in K} \sum_{w \in WM} \sum_{i \in CD} 
    c^{tr} \cdot d^{CD,WM}_{i,w} \cdot y^{WM}_{k,w,i,t}
\Bigg]
\]

\vspace{0.3cm}

\textbf{Inter-Facility Transportation Cost:}
\[
\sum_{t \in T}
\left[
    \sum_{k \in K} \sum_{i \in CD} \sum_{j \in DS}
    c^{tr} \cdot d^{CD,DS}_{i,j} \cdot y^{CD,DS}_{k,i,j,t}
\right]
\]

\vspace{0.3cm}

\textbf{Last-Mile Delivery Cost:}
\[
\sum_{t \in T} \sum_{f \in F}
c^{del}\,\bar{d}_f\,h_{f,t}
\left(\bar{h}\sum_{k \in K} q_k\right)
\]

\vspace{0.3cm}

\textbf{Mobile Distribution Center Movement Cost:}
\[
\sum_{t \in T} \sum_{z \in \mathcal{Z}} c^{mov}\, g(z)\, x_z
\]

\vspace{0.5cm}
